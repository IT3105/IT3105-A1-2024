// Function to change background image based on scroll position
function updateBackgroundImage() {
    const scrollPosition = window.scrollY;
    const windowHeight = window.innerHeight;

    if (scrollPosition < windowHeight) {
        document.body.style.backgroundImage = "url('genshin2.jpg')";
    } else if (scrollPosition < windowHeight * 2) {
        document.body.style.backgroundImage = "url('Pillars.jpg')";
    } else {
        document.body.style.backgroundImage = ""; // Default background if needed
    }

    // Ensure the new background images also cover the viewport
    document.body.style.backgroundSize = "cover";
    document.body.style.backgroundAttachment = "fixed"; // Make background constant and fixed
}

// Function to show subscription details based on the selected service
function showSubscriptionDetails(service) {
    const detailsDiv = document.getElementById('subscription-details');
    let detailsHtml = '';

    switch(service) {
        case 'netflix':
            detailsHtml = `
                <h2>Thank you for subscribing to Netflix!</h2>
                <p>Enjoy unlimited movies and TV shows with Netflix. Your subscription is active, and you can start streaming now!</p>
            `;
            break;
        case 'spotify':
            detailsHtml = `
                <h2>Thank you for subscribing to Spotify!</h2>
                <p>Stream unlimited music and podcasts with Spotify. Your subscription is active, and you can start listening now!</p>
            `;
            break;
        default:
            detailsHtml = `
                <h2>Subscription Error</h2>
                <p>Sorry, something went wrong. Please try again.</p>
            `;
            break;
    }

    detailsDiv.innerHTML = detailsHtml;
    detailsDiv.style.display = 'block'; // Show the details
}

// Add scroll event listener
window.addEventListener("scroll", updateBackgroundImage);
